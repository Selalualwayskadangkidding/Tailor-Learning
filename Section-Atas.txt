    (function () {
      const el = document.getElementById("socialProof");
      const avatarEl = document.getElementById("spAvatar");
      const titleEl = document.getElementById("spTitle");
      const subEl = document.getElementById("spSub");
      const closeBtn = document.getElementById("spClose");

      const messages = [
        { name: "Dina", title: "Dina sudah memesan", sub: "beberapa menit yang lalu" },
        { name: "Rani", title: "Rani baru ambil Paket Pro", sub: "3 menit yang lalu" },
        { name: "Mila", title: "Mila sudah memesan", sub: "5 menit yang lalu" },
        { name: "Sari", title: "Sari baru ambil Paket Basic", sub: "7 menit yang lalu" },
        { name: "Nina", title: "Nina sudah memesan", sub: "10 menit yang lalu" },
        { name: "Ayu", title: "Ayu baru ambil Paket Pro", sub: "12 menit yang lalu" }
      ];

      let idx = 0;
      let isClosed = false;

      function show(msg) {
        const initial = (msg.name || msg.title || "?").trim().charAt(0) || "?";
        avatarEl.textContent = initial.toUpperCase();
        titleEl.textContent = msg.title;
        subEl.textContent = msg.sub;
        el.classList.add("show");
      }

      function hide() {
        el.classList.remove("show");
      }

      function cycle() {
        if (isClosed) return;

        show(messages[idx % messages.length]);
        idx++;

        setTimeout(() => {
          hide();
          setTimeout(() => cycle(), 3500);
        }, 4000);
      }

      closeBtn.addEventListener("click", () => {
        isClosed = true;
        hide();
      });

      setTimeout(cycle, 1200);
    })();
  </script>

  <script>
    (function () {
      const track = document.getElementById('testiTrack');
      const slider = document.getElementById('testiSlider');
      const dotsWrap = document.getElementById('testiDots');
      if (!track || !slider || !dotsWrap) return;

      const slides = Array.from(track.children);
      const total = slides.length;

      let index = 0;
      let timer = null;
      const intervalMs = 3000;

      const dots = slides.map((_, i) => {
        const b = document.createElement('button');
        b.type = 'button';
        b.setAttribute('aria-label', 'Ke testimoni ' + (i + 1));
        b.addEventListener('click', () => {
          goTo(i);
          restart();
        });
        dotsWrap.appendChild(b);
        return b;
      });

      function updateDots() {
        dots.forEach((d, i) => d.classList.toggle('active', i === index));
      }

      function goTo(i) {
        index = (i + total) % total;
        track.style.transform = `translateX(-${index * 100}%)`;
        updateDots();
      }

      function next() {
        goTo(index + 1);
      }

      function start() {
        stop();
        timer = setInterval(next, intervalMs);
      }

      function stop() {
        if (timer) clearInterval(timer);
        timer = null;
      }

      function restart() {
        start();
      }

      slider.addEventListener('mouseenter', stop);
      slider.addEventListener('mouseleave', start);
      slider.addEventListener('touchstart', stop, { passive: true });
      slider.addEventListener('touchend', start);

      goTo(0);
      start();
    })();
  </script>
